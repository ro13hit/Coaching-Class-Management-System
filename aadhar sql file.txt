CREATE TABLE IF NOT EXISTS NEWST
(REGNO INT(5) PRIMARY KEY,
DOA DATE NOT NULL,
NAME VARCHAR(30) NOT NULL,
CLASS INT(2) NOT NULL,
DOB DATE NOT NULL,
FATHERS_NAME VARCHAR(30) NOT NULL,
SCHOOL VARCHAR(30) NOT NULL,
PHONE INT(10) NOT NULL,
ADDRESS VARCHAR(50) NOT NULL);

CREATE TABLE IF NOT EXISTS DEMOC
(DREGN INT(5) PRIMARY KEY,
NAME VARCHAR(30) NOT NULL,
CLASS INT(2) NOT NULL,
PHONE INT(10) NOT NULL,
SCHOOL VARCHAR(50) NOT NULL,
ADDRESS VARCHAR(60) NOT NULL);

CREATE TABLE IF NOT EXISTS FEECH
(NAME VARCHAR(30) NOT NULL,
PHONE VARCHAR(12) NOT NULL,
CHEQUE_NO VARCHAR(30) PRIMARY KEY,
BANK_NAME VARCHAR(50) NOT NULL,
DATE DATE NOT NULL,
AMOUNT INT(6) NOT NULL);

CREATE TABLE IF NOT EXISTS TEACHERS
(NAME VARCHAR(30) NOT NULL,
GRADE CHAR(1) NOT NULL,
PHONE VARCHAR(12) NOT NULL,
CITY VARCHAR(30) NOT NULL,
TEACHER_CODE INT PRIMARY KEY);
